-- docker部署
docker run -d \
  --name stock-ocr-api \
  -p 8002:8002 \
  -v "/Users/yangyang/vscode:/data" \

  docker logs -f stock-ocr-api

  docker ps

  docker stop stock-ocr-api

-- 删除
  docker rm -f stock-ocr-api
